def plate_number
  case plate_or_tube_name
    when /^M(\d+)$/  then $1.to_i
    when /^.*_(\d+)$/ then $1.to_i
    else 0
  end
end

def well_coord
  well_alpha = WELL_LETTER[(plate_position - 1)/12]
  well_num   = (plate_position - 1) % 12 + 1 
  return (plate_or_tube_name[0,1] == 'T' ? ' ' : well_alpha + well_num.to_s)
end
